cmake_minimum_required(VERSION 2.6)

project(citra)

SET(GCC_COMPILE_FLAGS "-std=c++11 -fpermissive")

# silence some spam
add_definitions(-Wno-attributes)
add_definitions(-DSINGLETHREADED)
add_definitions(${GCC_COMPILE_FLAGS})

# dependency checking
set(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} ${CMAKE_SOURCE_DIR}/CMakeTests)
include(FindOpenGL REQUIRED)
include(FindX11 REQUIRED)
find_package(PkgConfig REQUIRED)
pkg_search_module(GLFW REQUIRED glfw3)

include_directories(${GLFW_INCLUDE_DIRS})
option(DISABLE_QT5 "Disable Qt5 GUI" OFF)
if(NOT DISABLE_QT5)
    include(FindQt5)
    find_package(Qt5 COMPONENTS QtCore QtGui QtWidgets QtOpenGL)

    if(QT5_FOUND AND QT_QTCORE_FOUND AND QT_QTGUI_FOUND AND QT_QTWIDGETS_FOUND AND QT_QTOPENGL_FOUND)
        include(${QT_USE_FILE})
        include_directories(${QT_INCLUDES})
        include_directories(externals/qhexedit)
    else()
        message("Qt5 libraries not found! Disabling Qt5 GUI")
    endif()
endif()

# generate git revision information
list(APPEND CMAKE_MODULE_PATH "${CMAKE_CURRENT_SOURCE_DIR}/externals/cmake-modules/")
include(GetGitRevisionDescription)
get_git_head_revision(GIT_REF_SPEC GIT_REV)
git_describe(GIT_DESC --always --long --dirty)
git_branch_name(GIT_BRANCH)

# external includes
include_directories(${OPENGL_INCLUDE_DIR})
    
# internal includes
include_directories(src)

# process subdirectories
if(QT5_FOUND AND QT_QTCORE_FOUND AND QT_QTGUI_FOUND AND QT_QTWIDGETS_FOUND AND QT_QTOPENGL_FOUND AND NOT DISABLE_QT5)
    add_subdirectory(externals/qhexedit)
endif()
add_subdirectory(src)
